#!/bin/bash
# 定义样本列表（根据实际名称修改）
samples=("BG_hap1" "CF_hap1" "CM_hap1" "MRZ_hap1" "BG_hap2" "CF_hap2" "CM_hap2" "MRZ_hap2") # 替换为你的样本名
# 定义映射质量阈值（根据需求修改）
MAPQ=32
# 循环提交任务
for sample in "${samples[@]}"; do
    echo "正在提交任务: ${sample}"
    samtools sort -@ ${MAPQ} -o "${sample}.sorted.bam" "${sample}.sam"
    nohup samtools view -bh -u -F0xF0C -q ${MAPQ} "${sample}.sorted.bam" | \
    bedtools bamtobed | \
    awk -v OFS='\t' '{$4=substr($4,1,length($4)-2); print}' > "${sample}.bed" 2>&1 &
done
echo "所有任务已提交到后台！"
