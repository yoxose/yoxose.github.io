import re
#读入文件A(Read file A)
with open('C:\\Users\\yoxose\\Desktop\\testA.txt', 'r') as file:
    file_A = file.read()
print(file_A)
#读入文件B(Read file B)
with open('C:\\Users\\yoxose\\Desktop\\testB.txt', 'r') as file:
    file_B = file.read()
print(file_B)

#选择文件中特定的字符串(Selecting a specific string in a file)
pattern = r'Homo_(.*?):'
matches = re.findall(pattern, file_A)
print(matches)
pattern1 = r'Homo_(.*?)_'
matches1 = re.findall(pattern1, file_B)
print(matches1)


#将要编辑的文件A的内容赋值给定义的变量(The contents of file A to be edited are assigned to the defined variable)
file_content = file_A

#通过循环寻找A、B文件中特定字符串相同的长字符串，将A中的长字符串替换为B中的相应字符串
#(Replace the long string in A with the corresponding string in B by looping through the files A and B to find the long string with the same specific string.)
for match in matches:
    extracted_text = match.strip()
    print(extracted_text)
    for match1 in matches1:
        extracted_text1 = match1.strip()
        print(extracted_text1)
        if extracted_text == extracted_text1:
            #提取B中的相应长字符串赋值给y(Extract the corresponding long string in B and assign it to y)
            with open('C:\\Users\\yoxose\\Desktop\\testB.txt', 'r') as file_b:
                lines_b = file_b.readlines()
            for line in lines_b:
                if extracted_text1 in line:
                    y = line.strip()
                    print(y)
                    break 
            #print("Match found:")
            #print(extracted_text)
            print('yes')
            #替换A中的长字符串为B中的长字符串(Replace long string in A with long string in B)
            modified_contents = re.sub(rf'{re.escape(extracted_text)}:', f'I{y}:', file_content)
            #保存本次循环结果，下次循环在本循环结果的基础上修改(Save the result of this loop, and the next loop will be modified based on the result of this loop.)
            file_content = modified_contents
        else:
            print('no')
#删除特定字符I>前18个字符(Delete specific characters I> first 18 characters)
result_str = re.sub(r'.{18}I>', 'I>', file_content)
#删除I>(Delete I>)
result = result_str.replace('I>', '')
#保存最后结果(save)
with open('a.txt', 'w') as file_a:
    file_a.write(result)
