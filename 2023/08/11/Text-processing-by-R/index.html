library(foreign)
library(readxl)
data1 <- read_excel(“111.xlsx”)
data2 <- read_excel(“222.xlsx”)

character_data2 <- as.character(data2$name)

i=1
for(i in 1:length(data1$name)){
    a <- as.character(data1[i,1]) #Read the data in the first row and first column of data1
    b <- strsplit(a, " ") #Separate them with a space separator
    b <- unlist(b) #Convert the list to characters
    c <- grep('w',b,value=T,invert=F) #Matches a string with a w
    data1[i,2] <- grep(c,character_data2,value=T) #Matches the string in data2 with the value assigned to c in the previous step and outputs it
}
