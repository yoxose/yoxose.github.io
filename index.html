import os
import re
#自定义函数
def split_line(text, keyword):
    f = open(text, "r", encoding="utf-8")
    content = f.read()
    pattern = re.compile(keyword)#正则表达式找到特定字符
    split_text = pattern.sub(r"\n" + keyword, content)#替换特定字符
    return split_text

text = os.path.join("F:\\Python_code\\Programs\\edit_txt\\pedicularis(1)", "focname.txt")
new_text = split_line(text, r"Pedicularis")
print(new_text)

def split_line2(text, keyword):
    pattern = re.compile(keyword)
    split_text = pattern.sub(keyword + r"\n", text)
    return split_text

new_text2 = split_line2(new_text, r"马先蒿")
print(new_text2)

type(new_text2)
lines = new_text2.splitlines()
print(lines)
new_lines = []
for line in lines:
    if "Pedicularis" in line or "马先蒿" in line:
    pattern = re.compile(r"\?t=z\">")#注意特殊符号要在其前方加上\进行转义
    line = pattern.sub("\t", line)
    new_lines.append(line)
    print(line)
print(new_lines)
type(new_lines)

with open("focnameoutput.txt", "w", encoding="utf-8") as f:
for line in new_lines:
    f.write(line + "\n")#\n是分行

#将文件夹及子文件生成目录树
import sys
from pathlib import Path

tree_str = ''

def generate_tree(pathname, n=0):
    global tree_str
    if pathname.is_file():
        tree_str += ' |' * n + '-' * 4 + pathname.name + '\n'
    elif pathname.is_dir():
        tree_str += ' |' * n + '-' * 4 + \
            str(pathname.relative_to(pathname.parent)) + '\\' + '\n'
        for cp in pathname.iterdir():
            generate_tree(cp, n+1)

def save_file(tree, filename='tree.txt'):
    with open(filename, 'w', encoding='utf-8') as f:
        f.write(tree)

if __name__ == '__main__':
        # 命令参数个数为2并且目录存在存在
    if len(sys.argv) == 2 and Path(sys.argv[1]).exists():
        generate_tree(Path(sys.argv[1]), 0)
    # 命令参数个数为3并且目录存在存在
    if len(sys.argv) == 3 and Path(sys.argv[1]).exists():
        generate_tree(Path(sys.argv[1]), 0)
        save_file(tree_str, sys.argv[2])
    else:  # 当前路径
        generate_tree(Path.cwd(), 0)
    print(tree_str)
